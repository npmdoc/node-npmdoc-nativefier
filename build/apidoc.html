function nativefier.inferUserAgent.getUserAgentString (chromeVersion, platform)

  • description and source-code
    function getUserAgentString(chromeVersion, platform) {
    var userAgent = void 0;
    switch (platform) {
        case 'darwin':
            userAgent = 'Mozilla/5.0 (Macintosh; Intel Mac OS X 10_12_5) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/' + chromeVersion
 + ' Safari/537.36';
            break;
        case 'win32':
            userAgent = 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/' + chromeVersion
 + ' Safari/537.36';
            break;
        case 'linux':
            userAgent = 'Mozilla/5.0 (X11; Linux x86_64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/' + chromeVersion + ' Safari
/537.36';
            break;
        default:
            throw 'Error invalid platform specified to getUserAgentString()';
    }
    return userAgent;
}
